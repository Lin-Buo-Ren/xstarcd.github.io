screen

  • ~/.screenrc 
    #
# ~/.screenrc
#
# 启动时不显示欢迎屏幕
startup_message off

# 定义screen的功能键为Ctrl-A。向终端输入Ctrl-A时应按 Ctrl-A a。
#escape ^Aa

# 屏幕缓冲区 1024 行。
defscrollback 1024

# 显示窗口列表
caption always "%-w%{= BW}%50>%n %t%{-}%+w%<"

# 按 Ctrl-A w 或 Ctrl-A Ctrl-W 显示窗口列表
bind w windowlist -b
#bind ^w windowlist -b

#在使用screen过程中，经常会遇到“闪屏”的问题，这是可以关掉的。
#快速关掉：也就是先输入CTRL + a，再输入CTRL + g。
#永久关闭：修改/etc/screenrc 加入
vbell off
  • 新开标签页运行命令 
    # 在screen内部执行该命令
screen -x -X screen top

# 先建好会话TEST:Start as daemon: Screen session in detached mode.
screen -dmS TEST
# 在外面执行，会在screen会话中开一个top标签运行top
screen -S TEST -t top -x -X screen top
