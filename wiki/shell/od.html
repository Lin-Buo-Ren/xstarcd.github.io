od

od - dump files in octal and other formats 

# 生成测试文件
echo abcdef >tmp.txt

# 以16进制显示，跳转1个字节，取3个字节
od -t x -c -j 1 -N 3 tmp.txt 

0000001  62  63  64
          b   c   d
0000004

# -t x1 以16进制显示
# -c 显示ASCII码
# -j 跳过字节数
# -N 读取字节数
